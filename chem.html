function exportPDF() { // Hide sidebar before export const sidebar = document.querySelector('.sidebar'); const sidebarOriginalDisplay = sidebar ? sidebar.style.display : 'visible'; if (sidebar) sidebar.style.display = 'none'; try { const reportRoot = buildPrintableReport(); if (!reportRoot) { throw new Error('Report could not be generated'); } document.body.appendChild(reportRoot); const opt = { margin: [5, 5, 5, 5], // margins: [top, left, bottom, right] in mm filename: 'health_report.pdf', image: { type: 'jpeg', quality: 0.98 }, html2canvas: { scale: 2, useCORS: true, allowTaint: true, backgroundColor: '#ffffff', logging: false, windowHeight: 1200 }, jsPDF: { unit: 'mm', format: 'a4', orientation: 'landscape', // Landscape for wide tables compress: true }, pagebreak: { mode: ['css', 'legacy'], before: ['section'], after: ['section'] } }; html2pdf() .set(opt) .from(reportRoot) .save() .then(() => { reportRoot.remove(); if (sidebar) sidebar.style.display = sidebarOriginalDisplay; console.log('✅ PDF exported successfully'); }) .catch(error => { console.error('❌ Error exporting PDF:', error); reportRoot.remove(); if (sidebar) sidebar.style.display = sidebarOriginalDisplay; alert('ไม่สามารถ export PDF ได้: ' + error.message); }); } catch (error) { console.error('❌ Error in exportPDF:', error); if (sidebar) sidebar.style.display = sidebarOriginalDisplay; alert('เกิดข้อผิดพลาด: ' + error.message); } }
